## Instalação prometheus node-exporter grafana

```bash
# Criar diretório e navegar até ele
mkdir -p /root/monitoring
cd /root/monitoring

# Criar o arquivo docker-compose.yml com os serviços do Prometheus, Node Exporter e Grafana
cat <<EOF > docker-compose.yml
version: '3.7'

services:
  prometheus:
    image: prom/prometheus
    container_name: prometheus
    volumes:
      - ./prometheus.yml:/etc/prometheus/prometheus.yml
      - ./prometheus:/prometheus
    ports:
      - "9091:9090"  # Porta 9091 para evitar conflito
    networks:
      - monitoring_net
    restart: unless-stopped

  node-exporter:
    image: prom/node-exporter
    container_name: node-exporter
    ports:
      - "9101:9100"  # Porta 9101 para evitar conflito
    networks:
      - monitoring_net
    restart: unless-stopped
  
  grafana:
    image: grafana/grafana
    container_name: grafana
    volumes:
      - ./grafana:/var/lib/grafana
    environment:
      - GF_SECURITY_ADMIN_PASSWORD=admin
    ports:
      - "3001:3000"  # Porta 3001 para evitar conflito
    networks:
      - monitoring_net
    restart: unless-stopped

networks:
  monitoring_net:
    driver: bridge
EOF

# Iniciar os serviços com Docker Compose
sudo docker compose up -d

```
